Funkcionálne programovanie

  • Nemeniteľné dátové štruktúry / Vedľajšie efekty
  • Funkcie vyššieho rádu
  • funkcionalne programovanie ako ukazeme je riesenim pre mnohe vyzvy clean code

// const sampleData = ...
// Request 1: log the data to console
// 1. option
for (var i = 0; i < sampleData.length; i++) {
   console.log(sampleData[i]);
}
// 2. option
console.log(sampleData); // Why this works?
// 3. option
sampleData.forEach(console.log); // How this works?
				

// Request 2: make the log pretty
for (var i = 0; i < sampleData.length; i++) {
   console.log(sampleData[i].first_name + ' '
            + sampleData[i].last_name + ' from '
            + sampleData[i].city + ', '
            + sampleData[i].state);
}
				

// Request 2: make the log pretty
function personPrettyPrint(person) {
   return person.first_name + ' '
            + person.last_name + ' from '
            + person.city + ', '
            + person.state;
}
// 1. option
for (var i = 0; i < sampleData.length; i++) {
   console.log(personPrettyPrint(sampleData[i]));
}
// 2. option
console.log(personPrettyPrint(sampleData));
//Why this doesn't work?
//How should we change it to make it work?
				

// Request 2: make the log pretty
function personPrettyPrint(person) {
   return person.first_name + ' '
            + person.last_name + ' from '
            + person.city + ', '
            + person.state;
}
// 3. option
sampleData.map(personPrettyPrint).forEach(console.log);
// or
sampleData.map(personPrettyPrint).forEach((str) => console.log(str));
				

// Request 3: show only people with first name starting with A
sampleData
	.filter((person) => person.first_name.startsWith('A'))
	.map(personPrettyPrint)
	.forEach((str) => console.log(str));
